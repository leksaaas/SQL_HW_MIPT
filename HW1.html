import scipy.linalg #  
def f(x):
    return np.sin(x/5) * np.exp(x/10) + 5*np.exp(-x/2)
def ap(xi): # xi -  ,   ,   -   +1
    f_ap = np.array(list(map(f, xi)))
    A = []
    for x in xi:
        A.append([])
        for j in range(len(xi)):
            A[-1].append(x**j)
    A = np.array(A)
    print(A)
    u = scipy.linalg.solve(A, f_ap)
    print(u)
    X = np.arange(1, 15, 0.1)
    F = []
    Px = []
    for x in X:
        F.append(f(x))
        Px.append(0)
        for j in range(len(u)):
            Px[-1] += u[j] * x**j #  ()
    F = np.array(F)
    Px = np.array(Px)
    plot.scatter(X, F) # 
    plot.scatter(X, Px) # 
    plot.show()
